иҝҷдёҖзҜҮе°ұи®ІдёүдёӘеӣһзӯ”иҝҷй—®йўҳзҡ„е·Ҙе…·:Pulumi(еӨҡдә‘ / зңҹд»Јз Ғ / и·ҹ Terraform еҗҢдёҖеҘ— Provider)гҖҒAWS CDK(дәҡй©¬йҖҠе®ҳж–№ / зңҹд»Јз Ғ / зј–иҜ‘еҲ° CloudFormation)гҖҒCrossplane(K8s еҺҹз”ҹ / жҠҠеҹәзЎҖи®ҫж–ҪеҸҳжҲҗ K8s иө„жәҗ)гҖӮдёүиҖ…еҗ„жңүзңҹе®һзҡ„дҪҝз”ЁеңәжҷҜ,д№ҹеҗ„жңү"еҲ«зһҺжҚў"зҡ„иҫ№з•ҢвҖ”вҖ”дёҚжҳҜж–°е°ұдёҖе®ҡеҘҪгҖӮ

дёҖеҸҘиҜқе…Ҳи®°дҪҸ:HCL дёҚеӨҹз”ЁжҳҜзңҹзҡ„з—ӣ,дҪҶ"зңҹзј–зЁӢиҜӯиЁҖеҶҷ IaC"д№ҹеёҰжқҘзңҹзҡ„д»Јд»·вҖ”вҖ”жҠҪиұЎиҝҮеәҰ / и°ғиҜ•еӨҚжқӮ / Audit йҡҫгҖӮTerraform зҡ„"з¬Ё"еңЁеӨ§еһӢеӣўйҳҹйҮҢжҳҜдёҖз§ҚдҝқжҠӨжңәеҲ¶вҖ”вҖ”е®ғејәиҝ«дҪ жҠҠеҹәзЎҖи®ҫж–ҪеҶҷеҫ—з®ҖеҚ•зӣҙзҷҪгҖӮPulumi / CDK з»ҷдҪ иҮӘз”ұ,дҪ иҰҒе…ҲиҜҒжҳҺдҪ еӣўйҳҹй©ҫй©ӯеҫ—дәҶиҝҷд»ҪиҮӘз”ұгҖӮ

  minSize: env === "prod" ? 3 : 1,

// RDS вҖ”вҖ” з”Ё Pulumi зҡ„ abstraction

2.4 Pulumi зҡ„зңҹе®һз—ӣзӮ№ вҖӢ

еҲ«еҸӘзңӢе№ҝе‘Ҡ,Pulumi д№ҹжңү Terraform жІЎжңүзҡ„з—ӣ:

  1. з”ҹжҖҒжҜ” Terraform е°ҸвҖ”вҖ”жӢӣиҒҳеёӮеңә / зӨҫеҢәж–ҮжЎЈ / Stack Overflow зӯ”жЎҲ / з¬¬дёүж–№ Module,Terraform йғҪжҳҜж•°йҮҸзә§дјҳеҠҝ
  2. дә§е“ҒжҲҗзҶҹеәҰе·®дёҖзӮ№вҖ”вҖ”жҹҗдәӣ Provider иЎҢдёәи·ҹ Terraform еҗҢжӯҘжңүе»¶иҝҹ,иҫ№зјҳзү№жҖ§дёҚдёҖе®ҡж”ҜжҢҒ
  3. жҠҪиұЎеұӮ"йӯ”жі•"еӨҡвҖ”вҖ”awsx.ec2.Vpc иҮӘеҠЁз®— subnet CIDR,иҮӘеҠЁеҲӣе»ә NAT,зңҒдәӢдҪҶд№ҹ"зңӢдёҚи§Ғ"гҖӮеҮәй—®йўҳж—¶и°ғиҜ•жҜ” HCL йҡҫ
  4. и°ғиҜ•ж Ҳж·ұвҖ”вҖ”TypeScript еҶҷзҡ„д»Јз Ғзј–иҜ‘жҲҗ Pulumi еҶ…йғЁиЎЁзӨәеҶҚи°ғ Provider,ж Ҳи·ҹиёӘдёҚзӣҙи§Ӯ
  5. дјҒдёҡзүҲжүҚжңүзҡ„еҠҹиғҪвҖ”вҖ”Policy Pack(еғҸ Sentinel)гҖҒSSOгҖҒAudit Log йғҪжҳҜж”¶иҙ№
  6. и·ЁиҜӯиЁҖеҚҸдҪңе·®вҖ”вҖ”еӣўйҳҹдёҖдёӘз”Ё TSгҖҒдёҖдёӘз”Ё Python,Pulumi йЎ№зӣ®жң¬иә«еҸӘиғҪйҖүдёҖз§Қ

typescript
// lib/network-stack.ts
import * as cdk from "aws-cdk-lib";
import * as ec2 from "aws-cdk-lib/aws-ec2";
import * as eks from "aws-cdk-lib/aws-eks";
import * as rds from "aws-cdk-lib/aws-rds";

export class NetworkStack extends cdk.Stack {
  constructor(scope: cdk.App, id: string, props: cdk.StackProps & { env: string }) {
    super(scope, id, props);

    const isProd = props.env === "prod";

    // VPC вҖ”вҖ” CDK зҡ„ abstraction жҜ” Pulumi иҝҳй«ҳ
    const vpc = new ec2.Vpc(this, "MainVpc", {
      ipAddresses: ec2.IpAddresses.cidr("10.0.0.0/16"),
      maxAzs: 3,
      natGateways: isProd ? 3 : 1,
      subnetConfiguration: [
        { name: "public",  subnetType: ec2.SubnetType.PUBLIC,   cidrMask: 24 },
        { name: "private", subnetType: ec2.SubnetType.PRIVATE_WITH_EGRESS, cidrMask: 24 },
        { name: "db",      subnetType: ec2.SubnetType.PRIVATE_ISOLATED, cidrMask: 24 },
      ],
    });

    // EKS вҖ”вҖ” дёҖиЎҢиө·дёҖдёӘз”ҹдә§еҸҜз”Ёзҡ„йӣҶзҫӨ
    const cluster = new eks.Cluster(this, "MainCluster", {
      vpc,
      version: eks.KubernetesVersion.V1_30,
      defaultCapacity: isProd ? 5 : 1,
      defaultCapacityInstance: isProd
        ? ec2.InstanceType.of(ec2.InstanceClass.M6I, ec2.InstanceSize.XLARGE)
        : ec2.InstanceType.of(ec2.InstanceClass.T3, ec2.InstanceSize.LARGE),
    });

    // RDS вҖ”вҖ” DatabaseInstance её®дҪ еӨ„зҗҶ SG / subnet / secrets
    const db = new rds.DatabaseInstance(this, "OrdersDb", {
      engine: rds.DatabaseInstanceEngine.postgres({ version: rds.PostgresEngineVersion.VER_16_2 }),
      vpc,
      vpcSubnets: { subnetType: ec2.SubnetType.PRIVATE_ISOLATED },
      instanceType: isProd
        ? ec2.InstanceType.of(ec2.InstanceClass.R6G, ec2.InstanceSize.XLARGE)
        : ec2.InstanceType.of(ec2.InstanceClass.T4G, ec2.InstanceSize.MEDIUM),
      allocatedStorage: 100,
      credentials: rds.Credentials.fromGeneratedSecret("admin"),   // иҮӘеҠЁз”ҹжҲҗиҝӣ Secrets Manager
      deletionProtection: isProd,
    });

    new cdk.CfnOutput(this, "VpcId", { value: vpc.vpcId });
    new cdk.CfnOutput(this, "DbEndpoint", { value: db.dbInstanceEndpointAddress });
  }
}

CDK дёҚеғҸ Pulumi / Terraform зӣҙжҺҘи°ғ API,е®ғзј–иҜ‘жҲҗ CloudFormation жЁЎжқҝ,еҶҚи®© CFN еҺ» applyгҖӮиҝҷеёҰжқҘеҘҪеӨ„д№ҹеёҰжқҘз—ӣ:

иҝҷе°ұжҳҜ Crossplane зҡ„ж ёеҝғеҚ–зӮ№вҖ”вҖ”з»ҹдёҖжҺ§еҲ¶е№ійқўгҖӮеҹәзЎҖи®ҫж–ҪеҸҳжӣҙе’Ңеә”з”ЁеҸҳжӣҙиө°е®Ңе…ЁзӣёеҗҢзҡ„ GitOps жөҒзЁӢ,иҝҷеҜ№"йҮҚ GitOps ж–ҮеҢ–"зҡ„еӣўйҳҹжҳҜе·ЁеӨ§зҡ„з®ҖеҢ–гҖӮ

4.4 Crossplane зҡ„зңҹе®һз—ӣзӮ№ вҖӢ

еҗ¬иө·жқҘеӨӘзҫҺеҘҪ,еқ‘д№ҹеҫҲеӨҡ:

  1. K8s жҳҜеҝ…йЎ»зҡ„вҖ”вҖ”жІЎжңү K8s е°ұжІЎ Crossplane,е°ҸйЎ№зӣ®дёҚдјҡдёәдәҶ IaC дёҠ K8s
  2. Provider жҲҗзҶҹеәҰеҸӮе·®дёҚйҪҗвҖ”вҖ”AWS Provider жҜ” GCP / Azure ејәеҫҲеӨҡ,е°Ҹдј—иө„жәҗ(еҰӮ AWS OrganizationsгҖҒMacie)з»ҸеёёжІЎиҰҶзӣ–
  3. зҠ¶жҖҒе…ЁеңЁ K8s etcdвҖ”вҖ”еҹәзЎҖи®ҫж–Ҫиө„жәҗеӨҡдәҶ etcd дјҡиҶЁиғҖ,etcd еҮәй—®йўҳеҪұе“Қж•ҙдёӘйӣҶзҫӨ
  4. Reconciliation ж…ўвҖ”вҖ”controller иҪ®иҜўжЁЎејҸ,дә‘иө„жәҗеҸҳжӣҙеҸҜиғҪиҰҒеҮ еҲҶй’ҹжүҚиў« detect
  5. еҲ  namespace иҜҜеҲ дә‘иө„жәҗвҖ”вҖ”kubectl delete ns orders дјҡеҲ жҺүйӮЈдёӘ namespace дёӢжүҖжңү CR,иҝӣиҖҢеҲ жҺүзңҹе®һдә‘иө„жәҗ(RDS жІЎдәҶ)гҖӮз”ҹдә§еңәжҷҜеҝ…йЎ»й…Қ deletionPolicy: orphan
  6. Composition еӯҰд№ жӣІзәҝйҷЎвҖ”вҖ”patches / transforms / pipelines иҝҷеҘ—иҜӯжі•жҜ” HCL иҝҳйҡҫжҮӮ
  7. и·ЁйӣҶзҫӨеҹәзЎҖи®ҫж–Ҫйҡҫз®ЎвҖ”вҖ”дёҖдёӘ K8s йӣҶзҫӨз®ЎиҮӘе·ұзҡ„иө„жәҗиҝҳиЎҢ,з®Ў"йӣҶзҫӨд№Ӣй—ҙзҡ„еҹәзЎҖи®ҫж–Ҫ"(network peering)еҸҳжүӯжӣІ

4.5 дҪ•ж—¶дёҠ Crossplane вҖӢ

йҖӮеҗҲ:
  - е·Із»ҸйҮҚеәҰз”Ё K8s + GitOps(ArgoCD / Flux)
  - е№іеҸ°еӣўйҳҹиҰҒз»ҷејҖеҸ‘иҖ…жҠҪиұЎеҮә"иҮӘеҠ©ејҸеҹәзЎҖи®ҫж–Ҫ"(PaaS зұ»еһӢе№іеҸ°)
  - еӣўйҳҹжҺҘеҸ—"K8s жҳҜжҺ§еҲ¶е№ійқў"иҝҷдёӘеҝғжҷә
  - иө„жәҗз”ҹе‘Ҫе‘Ёжңҹи·ҹеә”з”Ёзҙ§еҜҶиҖҰеҗҲ(дёҖдёӘ namespace = дёҖеҘ—еҹәзЎҖи®ҫж–Ҫ)

дёҚйҖӮеҗҲ:
  - жІЎ K8s зҡ„йЎ№зӣ®
  - иө„жәҗж•°йҮҸеӨ§дҪҶи·ҹеә”з”Ёи§ЈиҖҰ(дјҒдёҡзә§зҪ‘з»ң / IAM / еҗҲи§„иө„жәҗ)вҶ’ Terraform жӣҙйҖӮеҗҲ
  - еӣўйҳҹеҜ№ K8s CRD дёҚзҶҹ вҶ’ еӯҰд№ жӣІзәҝеҗғдёҚж¶Ҳ
  - и·ЁеӨҡдә‘йңҖиҰҒдё°еҜҢ Provider вҶ’ Terraform з”ҹжҖҒе®Ңиғң

дёҖдёӘзңҹе®һеңәжҷҜ:жңүеӣўйҳҹз”Ё Crossplane з»ҷејҖеҸ‘иҖ…жҸҗдҫӣ"иҮӘеҠ©ејҸ RDS"вҖ”вҖ”ејҖеҸ‘иҖ…жҸҗ kind: PostgreSQLInstance,еҮ еҲҶй’ҹжӢҝеҲ°дёҖдёӘ PGгҖӮиҝҷжҳҜ Crossplane жңҖй—Әе…үзҡ„з”ЁдҫӢгҖӮдҪҶиҝҷдёӘеӣўйҳҹзҡ„еҹәзЎҖ VPC / Transit Gateway / IAM иҝҳжҳҜз”Ё TerraformвҖ”вҖ”еӣ дёәиҝҷдәӣ"йӣҶзҫӨеӨ–"зҡ„иө„жәҗз”Ё Crossplane еҸҚиҖҢеҲ«жүӯгҖӮ

дә”гҖҒдёүиҖ…йҖүеһӢзҹ©йҳө вҖӢ

жҠҠеҶізӯ–еҒҡжҲҗдёҖеј иЎЁ,зӣҙжҺҘжҠ„иө°з”Ё:

дҪ зҡ„жғ…еҶөжҺЁиҚҗ
е·Іжңү Terraform дё”жІЎз—ӣзӮ№еҲ«зһҺжҚўвҖ”вҖ”иҝҒз§»жҲҗжң¬ > ж”¶зӣҠ
еӣўйҳҹ TypeScript / Python ејә,IaC йҖ»иҫ‘еӨҚжқӮPulumi
е…Ё AWS,жғіиҰҒ"L2 жҠҪиұЎ"зңҒдәӢAWS CDK
йҮҚ GitOps + K8s,жғіз»ҷејҖеҸ‘иҖ…иҮӘеҠ©е№іеҸ°Crossplane
еӨҡдә‘ + жӢӣиҒҳе®№жҳ“ + еӣўйҳҹ SRE ж–ҮеҢ–Terraform
ж–°йЎ№зӣ®,еӣўйҳҹ 5 дәәд»ҘдёӢTerraform(зӨҫеҢәиө„жәҗжңҖеҺҡ)
ж–°йЎ№зӣ®,еӣўйҳҹ 30 дәәд»ҘдёҠ,жңүејәе·ҘзЁӢж–ҮеҢ–Terraform / Pulumi йғҪиЎҢ,зңӢиҜӯиЁҖеҒҸеҘҪ
жғіз”Ёзңҹзј–зЁӢиҜӯиЁҖ + AWS еҚ•дә‘CDK > Pulumi(CDK жҠҪиұЎжӣҙй«ҳ)
жғіз”Ёзңҹзј–зЁӢиҜӯиЁҖ + еӨҡдә‘Pulumi > CDK
е·Із»ҸеңЁ CFN,жғіеҚҮзә§CDK(еҗҢз”ҹжҖҒеҚҮзә§)
е·Із»ҸеңЁ CloudFormation YAML еҸ—дёҚдәҶCDK

дёүиҖ…ж··з”Ёзҡ„еңәжҷҜд№ҹеҫҲеёёи§Ғ:

еҹәзЎҖи®ҫж–Ҫ(VPC / йӣҶзҫӨ / IAM)    вҶ’ Terraform
еә”з”ЁеұӮ K8s иө„жәҗ + еә”з”Ёзӣёе…ідә‘иө„жәҗ вҶ’ Crossplane via ArgoCD
еҚ•дә‘еҶ…дёҡеҠЎжҠҪиұЎиө„жәҗ              вҶ’ CDK(еҰӮжһңе…Ё AWS)
                              вҶ’ Pulumi(еҰӮжһңи·Ёдә‘)

жІЎжңү"еҝ…йЎ»еҸӘйҖүдёҖдёӘ"зҡ„и§„е®ҡгҖӮдҪҶжҜҸеӨҡдёҖз§Қ IaC е·Ҙе…·,еӣўйҳҹе°ұеӨҡдёҖд»ҪеӯҰд№ жҲҗжң¬е’Ңз»ҙжҠӨжҲҗжң¬вҖ”вҖ”еҠ д№ӢеүҚй—®дёҖеҸҘ:"жҲ‘еӣўйҳҹзңҹзҡ„жүӣеҫ—еҠЁдёӨеҘ—еҗ—?"

е…ӯгҖҒзңҹд»Јз ҒеҶҷ IaC зҡ„еҲ©дёҺејҠ вҖӢ

жҠҠеҘҪеӨ„е’ҢеқҸеӨ„жҳҺзЎ®еҶҷеҮәжқҘ,е…Қеҫ—еҗ¬е®ҢиҗҘй”ҖиҜқжңҜе°ұеӨҙи„‘еҸ‘зғӯ:

6.1 еҘҪеӨ„ вҖӢ

вң… еҫӘзҺҜ / жқЎд»¶ / жҠҪиұЎ вҖ”вҖ” еӨҚжқӮеңәжҷҜеҶҷиө·жқҘеғҸ"жӯЈеёёд»Јз Ғ"
вң… зұ»еһӢжЈҖжҹҘ вҖ”вҖ” IDE жҠҘй”ҷ,дёҚз”Ёзӯү plan жүҚеҸ‘зҺ°
вң… IDE и·іиҪ¬ / йҮҚжһ„ / иЎҘе…Ё вҖ”вҖ” ејҖеҸ‘ж•ҲзҺҮдёҠдёҖдёӘеҸ°йҳ¶
вң… еҚ•е…ғжөӢиҜ• вҖ”вҖ” jest / pytest / go test жөӢ IaC йҖ»иҫ‘
вң… и·ҹеә”з”Ёд»Јз ҒеҗҢж Ҳ вҖ”вҖ” жӢӣиҒҳ / Code Review / е·Ҙе…·й“ҫз»ҹдёҖ
вң… жҠҪиұЎйҮҚз”Ё вҖ”вҖ” иҮӘе·ұеҶҷ "Class" иҖҢдёҚжҳҜ module
вң… дёҡеҠЎйҖ»иҫ‘ вҖ”вҖ” "з»ҷжҜҸдёӘе®ўжҲ·иө·дёҖеҘ— stack" иҝҷз§ҚйңҖжұӮ,зңҹд»Јз ҒеҶҷжңҖжё…жҷ°

6.2 еқҸеӨ„ вҖӢ

вқҢ жҠҪиұЎиҝҮеәҰ вҖ”вҖ” L2 / L3 Construct её®дҪ еҒҡдәҶдёҖе ҶеҶіе®ҡ,еҮәй—®йўҳйҡҫи°ғ
вқҢ и°ғиҜ•еӨҚжқӮ вҖ”вҖ” TS вҶ’ Pulumi еҶ…йғЁ вҶ’ Provider вҶ’ дә‘ API,ж Ҳж·ұ
вқҢ Audit йҡҫ вҖ”вҖ” Terraform зҡ„ plan иҫ“еҮәе№ІеҮҖ,зңҹд»Јз Ғ IaC зҡ„ plan з»ҸеёёеёҰ"зңӢдјјж— е…і"зҡ„ diff
вқҢ State еӨҚжқӮеәҰеўһеҠ  вҖ”вҖ” Pulumi state / CFN stack жҜ” tfstate жӣҙй»‘зӣ’
вқҢ жӢӣиҒҳеёӮеңәзӘ„ вҖ”вҖ” Terraform е·ҘзЁӢеёҲжҜ” Pulumi е·ҘзЁӢеёҲеӨҡ 5 еҖҚ
вқҢ еӣўйҳҹй©ҫй©ӯй—Ёж§ӣй«ҳ вҖ”вҖ” зңҹд»Јз Ғ IaC еҶҷеҮә"еұҺеұұ"жҜ” HCL жӣҙйҡҗи”Ҫ
вқҢ й…ҚеҘ—з”ҹжҖҒејұ вҖ”вҖ” Atlantis / TF Cloud / зӨҫеҢә module йғҪеӣҙз»• Terraform иҪ¬

жңҖе…ій”®зҡ„дёҖзӮ№:HCL зҡ„"з¬Ё"еңЁеӨ§еӣўйҳҹжҳҜз§ҚдҝқжҠӨвҖ”вҖ”е®ғејәиҝ«дҪ еҶҷ"жүҒе№ігҖҒжҳҺжҳҫгҖҒе®№жҳ“ review"зҡ„д»Јз ҒгҖӮзңҹд»Јз Ғ IaC зҡ„иҮӘз”ұеәҰ,дјҡи®©дёҖдёӨдёӘй«ҳзә§е·ҘзЁӢеёҲеҶҷеҮә"е°Ҹз»„йҮҢжІЎдәәзңӢеҫ—жҮӮ"зҡ„жҠҪиұЎ,й•ҝжңҹз»ҙжҠӨеҷ©жўҰгҖӮ

дёҖдёӘеҸҚжЁЎејҸ:жҹҗеӣўйҳҹз”Ё Pulumi еҶҷдәҶеҘ—"еҹәзЎҖи®ҫж–Ҫ SDK",жҠҪиұЎдәҶ 6 еұӮ classгҖӮ3 дёӘжңҲеҗҺеҺҹдҪңиҖ…зҰ»иҒҢгҖӮж–°жқҘзҡ„дәәиҠұдёҖдёӘжңҲжүҚжҗһжҮӮ"жҲ‘иҰҒеҠ дёҖдёӘ SG и§„еҲҷ,иҜҘеңЁе“ӘдёӘ class йҮҢж”№"гҖӮиҝҷз§ҚдәӢз”Ё Terraform еӨ§жҰӮзҺҮдёҚдјҡеҸ‘з”ҹвҖ”вҖ”еӣ дёә HCL еҶҷдёҚеҮә 6 еұӮжҠҪиұЎгҖӮ

дёғгҖҒCrossplane е®һжҲҳ:е№іеҸ°е·ҘзЁӢзҡ„иҢғејҸ вҖӢ

Crossplane жңҖеҖјеҫ—еұ•ејҖзҡ„дёҚжҳҜ"е®ғжҖҺд№Ҳз”Ё",жҳҜе®ғд»ЈиЎЁзҡ„иҢғејҸвҖ”вҖ”е№іеҸ°е·ҘзЁӢгҖӮ

жІЎжңү Crossplane зҡ„дё–з•Ң:
  ејҖеҸ‘иҖ…иҰҒдёӘ RDS    вҶ’ жҸҗ IT е·ҘеҚ• вҶ’ SRE еҶҷ Terraform PR вҶ’ review вҶ’ apply вҶ’ йҖҡзҹҘејҖеҸ‘иҖ…
  3 еӨ©еҲ° 1 е‘Ё

жңү Crossplane зҡ„дё–з•Ң:
  ејҖеҸ‘иҖ…иҰҒдёӘ RDS    вҶ’ еҶҷ 1 дёӘ yaml вҶ’ kubectl apply вҶ’ 5 еҲҶй’ҹжӢҝеҲ°
  иҮӘеҠ©ејҸ

иҝҷе°ұжҳҜе№іеҸ°е·ҘзЁӢзҡ„ж ёеҝғзҗҶеҝө:е№іеҸ°еӣўйҳҹе°ҒиЈ…еӨҚжқӮеәҰ,ејҖеҸ‘иҖ…йҖҡиҝҮз®ҖеҚ• API иҮӘеҠ©гҖӮCrossplane дёҚжҳҜе”ҜдёҖе®һзҺ°иҝҷдёӘзҗҶеҝөзҡ„е·Ҙе…·,дҪҶе®ғжңҖиҙҙ K8s з”ҹжҖҒгҖӮ

еҜ№еә”еҲ°з»„з»ҮйҮҢзҡ„и§’иүІ:

е№іеҸ°еӣўйҳҹ (3-5 дәә):
  - з»ҙжҠӨ Crossplane е®үиЈ…
  - еҶҷ Composition / XRD(е®ҡд№ү"ејҖеҸ‘иҖ…иғҪз”Ёд»Җд№Ҳ API")
  - з»ҙжҠӨ Terraform з®Ўеә•еұӮ(VPC / IAM / йӣҶзҫӨжң¬иә«)
  - дҝқиҜҒжҠҪиұЎзҡ„е®үе…ЁжҖ§(Composition йҮҢй»ҳи®Өз§ҒзҪ‘гҖҒеҠ еҜҶгҖҒеӨҮд»Ҫ)

ејҖеҸ‘иҖ… (е…¶дҪҷжүҖжңүдәә):
  - kubectl apply иҮӘе·ұеә”з”Ё + иҮӘе·ұзҡ„еҹәзЎҖи®ҫж–Ҫ CR
  - дёҚйңҖиҰҒзҹҘйҒ“ RDS / S3 / Lambda жҖҺд№Ҳй…Қ
  - дёҚйңҖиҰҒ AWS жҺ§еҲ¶еҸ°жқғйҷҗ

иҝҷз§ҚеҲҶе·Ҙзҡ„еүҚжҸҗ:е№іеҸ°еӣўйҳҹзҡ„жҠҪиұЎиҰҒзңҹзҡ„е°ҒиЈ…дәҶеӨҚжқӮеәҰгҖӮеҰӮжһң Composition жҡҙйңІ 50 дёӘеӯ—ж®өз»ҷејҖеҸ‘иҖ…,йӮЈе°ұжҳҜжҠҠ Terraform module з”Ё K8s CRD йҮҚеҶҷдәҶдёҖйҒҚ,жІЎжңүи§ЈеҶід»»дҪ•й—®йўҳгҖӮ

еҘҪзҡ„ Crossplane Composition зҡ„ж Үеҝ—:

  • жҡҙйңІз»ҷејҖеҸ‘иҖ…зҡ„еӯ—ж®ө вүӨ 5 дёӘ
  • й»ҳи®Өй…ҚзҪ®е°ұжҳҜз”ҹдә§еҸҜз”Ё(з§ҒзҪ‘гҖҒеҠ еҜҶгҖҒеӨҮд»ҪгҖҒзӣ‘жҺ§)
  • ејҖеҸ‘иҖ…дёҚйңҖиҰҒжҮӮеә•еұӮдә‘зҡ„д»»дҪ•з»ҶиҠӮ
  • е№іеҸ°еҚҮзә§(ж”№ Composition й»ҳи®ӨеҖј)еҜ№ејҖеҸ‘иҖ…йҖҸжҳҺ

е…«гҖҒиё©еқ‘жҸҗйҶ’ вҖӢ

  1. д»Ҙдёә"зңҹд»Јз Ғ IaC е°ұдёҖе®ҡжҜ” Terraform еҘҪ"вҖ”вҖ”еӣўйҳҹй©ҫй©ӯеҫ—дәҶеҗ—?Audit и·ҹеҫ—дёҠеҗ—?жҠҪиұЎдёҚдјҡеӨұжҺ§еҗ—?
  2. Pulumi йЎ№зӣ®ж··з”ЁеӨҡз§ҚиҜӯиЁҖвҖ”вҖ”дёҖдёӘ stack з”Ё TS,еҸҰдёҖдёӘз”Ё Python,и·Ё stack еҚҸдҪңеҸҳжүӯжӣІ
  3. CDK з”Ё L3 construct дёҚзңӢеҶ…йғЁвҖ”вҖ”дёҖдёӘ ApplicationLoadBalancedFargateService иө·еҮә 30 дёӘиө„жәҗ,еҮәй—®йўҳдёҚзҹҘйҒ“д»Һе“ӘжҹҘ
