е№¶еҸ‘йҷ·йҳұпјҡз«һжҖҒжқЎд»¶дёҺжӯ»й”ҒеҲҶжһҗ вҖӢ

еҜјиҜ»пјҡGo е№¶еҸ‘ bug жңҖеҚұйҷ©зҡ„ең°ж–№дёҚжҳҜеҝ…зҺ°еҙ©жәғпјҢиҖҢжҳҜвҖңз»қеӨ§еӨҡж•°ж—¶еҖҷжӯЈеёёпјҢеҺӢеҠӣдёҖдёҠжқҘе°ұйҡҸжңәеҮәй”ҷвҖқгҖӮ

дёҖгҖҒе·ҘзЁӢеңәжҷҜпјҡеҒ¶еҸ‘й”ҷиҜҜжңҖйҡҫжҹҘ вҖӢ

дёҖдёӘи®Ўж•°еҷЁжң¬ең°еҺӢжөӢжІЎй—®йўҳпјҢдёҠзәҝеҗҺж•°жҚ®еҒ¶е°”е°‘еҮ жқЎпјӣдёҖдёӘжҺҘеҸЈе№іж—¶жӯЈеёёпјҢй«ҳеі°жңҹзӘҒз„¶еҚЎжӯ»пјӣдёҖдёӘд»»еҠЎжҳҺжҳҺе·Із»Ҹз»“жқҹпјҢиҝӣзЁӢйҮҢзҡ„ goroutine ж•°йҮҸеҚҙдёҖзӣҙдёҠж¶ЁгҖӮ

иҝҷдәӣй—®йўҳйҖҡеёёжқҘиҮӘдёүзұ»е№¶еҸ‘йҷ·йҳұпјҡ

  • з«һжҖҒжқЎд»¶пјҡеӨҡдёӘ goroutine иҜ»еҶҷеҗҢдёҖд»Ҫж•°жҚ®пјҢзјәе°‘еҗҢжӯҘгҖӮ
  • жӯ»й”ҒпјҡжүҖжңү goroutine йғҪеңЁзӯүеҲ«дәәе…ҲеҠЁгҖӮ
  • goroutine жі„жјҸпјҡgoroutine иў«ж°ёд№…йҳ»еЎһпјҢж— жі•йҖҖеҮәгҖӮ

е№¶еҸ‘д»Јз Ғзҡ„з¬¬дёҖеҺҹеҲҷжҳҜпјҡе…ұдә«зҠ¶жҖҒеҝ…йЎ»жңүжҳҺзЎ®зҡ„еҗҢжӯҘиҫ№з•ҢпјҢйҳ»еЎһж“ҚдҪңеҝ…йЎ»жңүжҳҺзЎ®зҡ„йҖҖеҮәи·Ҝеҫ„гҖӮ

дәҢгҖҒз«һжҖҒжқЎд»¶пјҡcounter++ дёҚжҳҜеҺҹеӯҗж“ҚдҪң вҖӢ

дёӢйқўзҡ„д»Јз ҒзңӢиө·жқҘз®ҖеҚ•пјҢдҪҶз»“жһңз»Ҹеёёе°ҸдәҺ 1000гҖӮ

go
package main

import (
    "fmt"
    "sync"
)

func main() {
    var counter int
    var wg sync.WaitGroup

    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            counter++
        }()
    }

    wg.Wait()
    fmt.Println(counter)
}

counter++ иҮіе°‘еҢ…еҗ«иҜ»еҸ–гҖҒеҠ дёҖгҖҒеҶҷеӣһеҮ дёӘжӯҘйӘӨгҖӮеӨҡдёӘ goroutine еҗҢж—¶жү§иЎҢж—¶пјҢеҪјжӯӨзҡ„еҶҷе…ҘдјҡиҰҶзӣ–гҖӮ

з”Ё race detector жЈҖжҹҘпјҡ

дёүгҖҒдҝ®еӨҚз«һжҖҒпјҡMutexгҖҒAtomicгҖҒChannel вҖӢ

1. з”Ё Mutex дҝқжҠӨе…ұдә«зҠ¶жҖҒ вҖӢ

go
package main

import (
    "fmt"
    "sync"
)

func main() {
    var counter int
    var mu sync.Mutex
    var wg sync.WaitGroup

    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            mu.Lock()
            counter++
            mu.Unlock()
        }()
    }

    wg.Wait()
    fmt.Println(counter)
}

еҰӮжһңдёҙз•ҢеҢәеҶ…еҸҜиғҪ return жҲ– panicпјҢдјҳе…ҲеҶҷжҲҗпјҡ

go
mu.Lock()
defer mu.Unlock()

2. з”Ё atomic еӨ„зҗҶз®ҖеҚ•и®Ўж•° вҖӢ

3. з”Ё channel дёІиЎҢеҢ–еҶҷе…Ҙ вҖӢ

go
updates := make(chan int)

go func() {

        total += n
    }
    fmt.Println(total)
    close(done)
}()

for i := 0; i < 1000; i++ {

3. AB-BA й”ҒйЎәеәҸ вҖӢ

go
var a sync.Mutex
var b sync.Mutex

func f1() {
    a.Lock()
    defer a.Unlock()
    b.Lock()
    defer b.Unlock()
}

func f2() {
    b.Lock()
    defer b.Unlock()
    a.Lock()
    defer a.Unlock()
}

дёӨдёӘ goroutine еҲҶеҲ«иҝӣе…Ҙ f1 е’Ң f2 ж—¶еҸҜиғҪдә’зӣёзӯүеҫ…гҖӮи§ЈеҶіж–№ејҸжҳҜжүҖжңүд»Јз Ғз»ҹдёҖеҠ й”ҒйЎәеәҸпјҢжҲ–иҖ…жҠҠдёӨдёӘй”ҒеҗҲе№¶дёәдёҖдёӘжӣҙзІ—зІ’еәҰзҡ„й”ҒгҖӮ

дә”гҖҒgoroutine жі„жјҸпјҡжІЎжңүйҖҖеҮәи·Ҝеҫ„ вҖӢ

жі„жјҸдёҚжҳҜеҶ…еӯҳзӣҙжҺҘжі„жјҸпјҢиҖҢжҳҜ goroutine ж°ёиҝңдёҚйҖҖеҮәпјҢиҝһеёҰдҝқз•ҷж ҲгҖҒеј•з”ЁеҜ№иұЎе’Ңиө„жәҗгҖӮ

еёёи§ҒдҫӢеӯҗпјҡ

go
func waitResult() <-chan string {
    ch := make(chan string)
    go func() {
        time.Sleep(2 * time.Second)
        ch <- "done"
    }()
    return ch
}

func main() {
    select {
    case v := <-waitResult():
        fmt.Println(v)
    case <-time.After(100 * time.Millisecond):
        fmt.Println("timeout")
    }
}

дё» goroutine и¶…ж—¶йҖҖеҮәеҗҺпјҢеҗҺеҸ° goroutine д»Қз„¶дјҡе°қиҜ•еҸ‘йҖҒеҲ° chпјҢдҪҶе·Із»ҸжІЎдәәжҺҘж”¶пјҢжңҖз»Ҳйҳ»еЎһгҖӮ

дҝ®еӨҚж–№ејҸжҳҜеј•е…Ҙ ContextпјҢе№¶дё”еҸ‘йҖҒж—¶д№ҹзӣ‘еҗ¬еҸ–ж¶Ҳпјҡ

go
func waitResult(ctx context.Context) <-chan string {
    ch := make(chan string)
    go func() {
        defer close(ch)
        select {
        case <-time.After(2 * time.Second):
        case <-ctx.Done():
            return
        }

        select {
        case ch <- "done":
        case <-ctx.Done():
            return
        }
    }()
    return ch
}

е…ӯгҖҒеҸҜиҝҗиЎҢжҺ’жҹҘзӨәдҫӢпјҡи§ӮеҜҹ goroutine ж•°йҮҸ вҖӢ

дёӢйқўзҡ„дҫӢеӯҗжЁЎжӢҹй”ҷиҜҜеҶҷжі•е’Ңдҝ®еӨҚеҶҷжі•пјҢиғҪзӣҙи§ӮзңӢеҲ° goroutine ж•°йҮҸзҡ„е·®ејӮгҖӮ

go
package main

import (
    "context"
    "fmt"
    "runtime"
    "time"
)

func main() {
    for i := 0; i < 1000; i++ {
        leak()
    }
    time.Sleep(200 * time.Millisecond)
    fmt.Println("after leak:", runtime.NumGoroutine())

    ctx, cancel := context.WithCancel(context.Background())
    for i := 0; i < 1000; i++ {
        noLeak(ctx)
    }
    cancel()
    time.Sleep(200 * time.Millisecond)
    fmt.Println("after cancel:", runtime.NumGoroutine())
}

func leak() {
    ch := make(chan int)
    go func() {
        ch <- 1
    }()
}

func noLeak(ctx context.Context) {
    ch := make(chan int)
    go func() {
        select {
        case ch <- 1:
        case <-ctx.Done():
            return
        }
    }()
}

1. д»Ҙдёә map е№¶еҸ‘иҜ»еҶҷе®үе…Ё вҖӢ

Go еҶ…зҪ® map дёҚжҳҜе№¶еҸ‘е®үе…Ёзҡ„гҖӮе№¶еҸ‘иҜ»еҶҷеҸҜиғҪ panicпјҡ

text

й«ҳйў‘иҜ»еҶҷеҸҜд»Ҙз”Ё sync.RWMutex дҝқжҠӨпјҢиҜ»еӨҡеҶҷе°‘дё” key зЁіе®ҡзҡ„еңәжҷҜеҸҜд»ҘиҖғиҷ‘ sync.MapгҖӮ

2. еӨҚеҲ¶еёҰй”Ғзҡ„з»“жһ„дҪ“ вҖӢ

еҢ…еҗ« sync.MutexгҖҒsync.WaitGroupгҖҒatomic зҡ„з»“жһ„дҪ“дёҚиҰҒеӨҚеҲ¶гҖӮеӨҚеҲ¶еҗҺй”ҒзҠ¶жҖҒе’Ңж•°жҚ®зҠ¶жҖҒдјҡеҲҶиЈӮпјҢиЎҢдёәеҫҲйҡҫйў„жөӢгҖӮ

3. WaitGroup Add ж—¶жңәй”ҷиҜҜ вҖӢ

wg.Add(1) еә”иҜҘеңЁеҗҜеҠЁ goroutine д№ӢеүҚи°ғз”ЁгҖӮеҗҰеҲҷ goroutine еҸҜиғҪе…Ҳжү§иЎҢ DoneпјҢеҜјиҮҙи®Ўж•°еҷЁеҸҳжҲҗиҙҹж•°гҖӮ

4. з”Ё time.After еҶҷй•ҝеҫӘзҺҜ вҖӢ

еңЁй«ҳйў‘еҫӘзҺҜйҮҢеҸҚеӨҚ time.After дјҡеҲӣе»әеӨ§йҮҸ timerгҖӮйңҖиҰҒеҸҜжҺ§еҒңжӯўж—¶пјҢз”Ё time.NewTimer жҲ– time.NewTicker е№¶дё»еҠЁ StopгҖӮ

е…«гҖҒз”ҹдә§еҲӨж–ӯ вҖӢ

  • go test -race ./... еҝ…и·‘пјҢе°Өе…¶жҳҜе…ұдә«зј“еӯҳгҖҒиҝһжҺҘз®ЎзҗҶгҖҒеҗҺеҸ°д»»еҠЎгҖӮ
  • жҜҸдёӘ goroutine йғҪиғҪиҜҙжё…жҘҡйҖҖеҮәжқЎд»¶гҖӮ
